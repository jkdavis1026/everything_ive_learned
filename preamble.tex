\tcbset{
  mybox/.style={
    colback=gray!10,   % light gray background
    colframe=black,    % black border
    boxrule=0.5pt,     % border thickness
    arc=1pt,           % rounded corners
    left=6pt, right=6pt, top=4pt, bottom=4pt,
    enhanced,
    sharp corners=south,   % makes top look flat if you add a title
    fonttitle=\bfseries,   % bold title (optional)
    coltitle=black,        % title text color
    title={},
  }
}

% Define a custom "IMPORTANT" box based on mybox
\tcbset{
  important/.style={
    mybox,                  % inherit from your mybox style
    colback=red!10,         % light red background
    colframe=red!60!black,  % darker red border
    title=IMPORTANT,        % default title
    fonttitle=\bfseries,    % bold title
  }
}
